faster-whisper

# CTranslate2 runtime (thường được kéo theo faster-whisper, liệt kê tường minh)
ctranslate2

# ── Speaker diarization ───────────────────────────────────────────────────────
pyannote.audio

# ── Deep learning framework ───────────────────────────────────────────────────
torch

# ── Hugging Face Hub (tải model về cache) ─────────────────────────────────────
huggingface-hub

# ── Audio / signal processing ─────────────────────────────────────────────────
# soundfile: đọc/ghi WAV không phụ thuộc libsndfile hệ thống
soundfile
# noisereduce: khử tiếng ồn nền (spectral gating) trước khi diarization
# Dùng với flag --denoise khi audio có noise stationary (văn phòng, quán cà phê)
# hoặc khi 2 giọng gần nhau hay bị nhầm speaker
# pip install noisereduce
noisereduce

# librosa: phân tích audio (tuỳ chọn – hữu ích nếu muốn inspect waveform)
# librosa>=0.10.0

# ── Utilities ─────────────────────────────────────────────────────────────────
# numpy: đã là transitive dep nhưng pin phiên bản để tránh ABI mismatch
numpy

# tqdm: thanh tiến trình (được dùng bởi pyannote & faster-whisper internally)
tqdm
